ESP32 Setup
Connect DHT22 + MQ-135 sensor hardware
Connect ESP32 Device
Enter the IP address shown in Serial Monitor after uploading the sketch below
IP Address
Port
Not connected — enter your ESP32 IP address above
Arduino Sketch — Upload this to your ESP32
#include <WiFi.h>
#include <WebServer.h>
#include <DHT.h>

// ── WiFi ──────────────────────────────
const char* ssid = "YOUR_SSID";
const char* password = "YOUR_PASSWORD";

#define DHT_PIN 4 // DHT22 DATA → GPIO 4
#define MQ135_PIN 34 // MQ-135 AOUT → GPIO 34
#define DHT_TYPE DHT22

DHT dht(DHT_PIN, DHT_TYPE);
WebServer server(80);

void handleData() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  int co2 = map(analogRead(MQ135_PIN), 0, 4095, 400, 2000);
  server.send(200, "application/json",
    "{\"temp\":" + String(t, 1) + ",\"hum\":" + String(h, 0) + ",\"co2\":" + String(co2) + "}");
}

void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  Serial.println(WiFi.localIP()); // ← Copy this IP to ThermIO
  server.on("/data", handleData);
  server.begin();
}
void loop() { server.handleClient(); }
Wiring Guide
DHT22 Sensor
VCC → 3.3V
GND → GND
DATA → GPIO 4
10kΩ resistor: DATA to VCC
MQ-135 Sensor
VCC → 5V
GND → GND
AOUT → GPIO 34
Allow 24–48 hr preheat
